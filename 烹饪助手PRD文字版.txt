连续烹饪助手产品需求文档（PRD）
1. 文档概述
1.1 项目名称
连续烹饪助手（Continuous Cooking Assistant）
1.2 文档目的
本文档旨在定义"连续烹饪助手"Web应用的产品需求，为开发团队提供详细的功能规格和设计指导。
1.3 项目背景
解决家庭烹饪中"采购成本高"和"多菜品制作流程混乱"的痛点，通过智能化采购和工业化烹饪编排提升烹饪效率。
2. 用户画像（Persona）

•年龄：30-45岁
•职业：企业职员/管理者
•痛点：工作繁忙，时间紧张，需要高效完成家庭烹饪

•年龄：25-40岁
•职业：自由职业者/创意工作者

3. 核心业务流程图（User Flow）
3.1 整体业务流程
用户登录 → 选择功能模块（采购/烹饪） → 执行核心操作 → 查看结果 → 反馈优化
3.2 采购模块流程
拍照识别小票/价签 → OCR提取信息 → 人工微调 → 存储到供应商数据库 → 生成价格仪表盘
3.3 烹饪编排模块流程
输入菜谱来源 → LLM解析菜谱 → 工序智能拆解 → 物料聚合 → 硬件配置识别 → 并行排程 → 生成甘特图 → 执行导航
3.4 烹饪执行流程
进入烹饪模式 → 大字体步骤导航 → 语音提醒关键节点 → 用户标记完成状态 → 动态调整时间轴

def aggregate_materials(tasks: List[CookingTask]) -> List[CookingTask]:
    """
    物料聚合算法
    """
    # 按食材名称和处理方式分组
    material_groups = {}
    for task in tasks:
        if task.type == 'prep':
            key = f"{task.ingredient.name}_{task.ingredient.preparation}"
            if key not in material_groups:
                material_groups[key] = []
            material_groups[key].append(task)
    
    # 合并相同物料处理任务
    aggregated_tasks = []
    for key, group_tasks in material_groups.items():
        if len(group_tasks) > 1:
            # 合并为一个任务，增加处理量
            merged_task = merge_tasks(group_tasks)
            aggregated_tasks.append(merged_task)
        else:
            aggregated_tasks.append(group_tasks[0])
    
    return aggregated_tasks

def optimize_parallel_schedule(tasks: List[CookingTask], 
                              dependencies: Dict[str, List[str]], 
                              resources: Dict[str, List[str]]) -> List[ScheduleItem]:
    """
    并行调度优化算法
    """
    # 初始化时间轴
    timeline = []
    current_time = 0
    
    # 按任务优先级排序（依赖少的优先）
    sorted_tasks = sort_by_dependency_priority(tasks, dependencies)
    
    # 任务调度循环
    for task in sorted_tasks:
        # 查找可用资源
        available_resource = find_available_resource(task, resources, timeline, current_time)
        
        if available_resource:
            # 分配资源并安排时间
            schedule_item = ScheduleItem(
                task_id=task.id,
                start_time=current_time,
                end_time=current_time + task.estimated_time,
                resource=available_resource
            )
            timeline.append(schedule_item)
            current_time = max(current_time, schedule_item.end_time)
        else:
            # 资源冲突，需要等待
            wait_time = find_next_available_time(task, resources, timeline)
            current_time += wait_time
            schedule_item = ScheduleItem(
                task_id=task.id,
                start_time=current_time,
                end_time=current_time + task.estimated_time,
                resource=find_available_resource(task, resources, timeline, current_time)
            )
            timeline.append(schedule_item)
            current_time = schedule_item.end_time
    
    return timeline

def find_available_resource(task: CookingTask, 
                           resources: Dict[str, List[str]], 
                           timeline: List[ScheduleItem], 
                           current_time: int) -> str:
    """
    查找可用硬件资源
    """
    # 根据任务类型查找可用资源
    if task.type == 'cook':
        stove_resources = resources.get('stoves', [])
        for stove in stove_resources:
            if is_resource_available(stove, timeline, current_time, current_time + task.estimated_time):
                return stove
    elif task.type == 'appliance':
        appliance_resources = resources.get('appliances', [])
        for appliance in appliance_resources:
            if is_resource_available(appliance, timeline, current_time, current_time + task.estimated_time):
                return appliance
    else:  # prep tasks can use any available resource
        all_resources = stove_resources + appliance_resources
        for resource in all_resources:
            if is_resource_available(resource, timeline, current_time, current_time + task.estimated_time):
                return resource
    
    return None

def detect_schedule_conflicts(schedule: List[ScheduleItem]) -> List[Conflict]:
    """
    检测时间轴冲突
    """
    conflicts = []
    
    # 检查资源冲突
    for i in range(len(schedule)):
        for j in range(i + 1, len(schedule)):
            if schedule[i].resource == schedule[j].resource:
                if has_time_overlap(schedule[i], schedule[j]):
                    conflicts.append(Conflict(
                        type='resource_conflict',
                        tasks=[schedule[i].task_id, schedule[j].task_id],
                        resource=schedule[i].resource
                    ))
    
    # 检查依赖冲突
    for task in schedule:
        dependencies = get_task_dependencies(task.task_id)
        for dep in dependencies:
            dep_schedule = find_schedule_item(dep, schedule)
            if dep_schedule and dep_schedule.end_time > task.start_time:
                conflicts.append(Conflict(
                    type='dependency_conflict',
                    tasks=[dep, task.task_id]
                ))
    
    return conflicts
